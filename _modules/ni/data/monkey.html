	"""
	Loads Data into a Panda Data Frame

	Use like this::

		data = ni.data.decoding_data.DecodingData()
		plot(data.data.ix[(0,0,0):(0,3,-1)].transpose().cumsum())

	.ix slices a portion of the data. The three indices are: Condition, Trial, then Cell. So the code above plots the cummulative firing rate for the first the trials of the first condition for all cells.


	.. image:: _static/decoding_data_slice.png

	or::

		data = ni.data.decoding_data.DecodingData()
		pylab.plot(data.data.xs(0,level='Condition').xs(0,level='Cell').ix[:5].transpose().cumsum())
		pylab.show()

	With .xs one can slice at any level, ie. get specific trials and cells. this can be combined with .ix to get slices of data.
	
	"""
	def __init__(self,trial_nr='101a03'):
		mat = scipy.io.loadmat(path + 'nic'+str(trial_nr)+'.mat')
		try:
			self.init_dot_indexing(mat)
		except:
			self.init_square_brackets_indexing(mat)

[docs]	def init_dot_indexing(self,mat):
		self.Spike_times_STC = mat['Data'][0][0].Spike_times_STC[0][0]
		self.T = mat['Data'][0][0].T[0][0]
		self.Stimulus = mat['Data'][0][0].Stimlus[0][0]
		self.name = mat['Data'][0][0].name
		self.nr_cells = np.size(self.Spike_times_STC.all_SUA[0][0].spike_times,2);
		self.nr_conditions = self.Spike_times_STC.all_SUA[0][0].Nr_stimlus[0][0]
		self.nr_trials = np.size(self.Spike_times_STC.all_SUA[0][0].spike_times,1);
		self.time_bins = 10000
		d = []
		tuples = []
		for con in range(self.nr_conditions):
			for t in range(self.nr_trials):
				for c in range(self.nr_cells):
						spikes = list(ni.model.pointprocess.getBinary(self.Spike_times_STC.all_SUA[0][0].spike_times[con,t,c].flatten()*1000))
						if spikes != []:
							d.append(spikes)
							tuples.append((con,t,c))
		index = pd.MultiIndex.from_tuples(tuples, names=['Condition','Trial','Cell'])
		self.data = pd.DataFrame(d, index = index)
		self.time_bins = len(self.data.columns)
