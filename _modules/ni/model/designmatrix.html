	def __init__(self,spikes,history_length=100,knot_number=5,order_flag=2,kernel=0):
		if np.any(kernel) == 0:
			self.history_kernel = cs.create_splines_logspace(history_length, knot_number, 0)
		else:
			self.history_kernel = kernel
		self.covariate_matrix, self.covariates, self.morder = cdm.create_design_matrix_vk(self.history_kernel, order_flag)
		self.matrix = scipy.signal.convolve2d(spikes, self.covariate_matrix)
		self.matrix = self.matrix[:len(spikes)]

		mult = int(np.ceil(self.length / splines.shape[0]))

			splines_full_length = np.tile(splines,(mult,1))
		self.matrix[:,self.width_used:(self.width_used+l)] = splines_full_length[:self.length]
