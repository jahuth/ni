def bootstrap(bootstrap_repetitions,model,data,test_data=[]):

[docs]	"""
		A helper function that performs bootstrap evaluation of models
	"""
	other_data = data.other_spikes
	#subjob("bootstrap")
	EICE = []
	EICE2 = []
	EIC_aic = []
	EIC_bic = []
	actual_fit = model.fit(data)
	llf_test_model = [actual_fit.predict(t)['LogLikelihood'] for t in test_data]
	llf_train = []
	llf_boot = []
	llf_test = []
	llf_test2 = []
	for boot_rep in range(bootstrap_repetitions):
		#dbg("running bootstrap "+str(boot_rep))
		#do_log(False)
		bootstrap_trials = np.array(np.floor(np.random.rand(int(np.ceil(data.nr_trials)))*data.nr_trials),int)
		boot_data = data.trial(bootstrap_trials)
		boot_data.other_spikes = other_data.trial(bootstrap_trials) # obsolete?
		boot_fit = model.fit(boot_data)

		boot_prediction = boot_fit.predict(data)
		model_prediction_fit = actual_fit.predict(data)
		model_prediction_boot = actual_fit.predict(boot_data)
		llf_train.append(boot_fit.fit.llf)
		llf_boot.append(boot_prediction['LogLikelihood'])
		llf_test.append([boot_fit.predict(t)['LogLikelihood'] for t in test_data])
		EICE.append(-2*actual_fit.fit.llf + 2*(boot_fit.fit.llf - model_prediction_boot['LogLikelihood'] + model_prediction_fit['LogLikelihood'] - boot_prediction['LogLikelihood']))
		EICE2.append(-2*actual_fit.fit.llf + 2*(boot_fit.fit.llf - boot_prediction['LogLikelihood']))
		# Knishi & Kitagawa 2007, p.198 Formula (8.23)
		EIC_aic.append(boot_fit.fit.aic)
		EIC_bic.append(boot_fit.fit.bic)
		#do_log(True)
	#superjob()
	return {'name': model.name, 'AIC':actual_fit.fit.aic,'EICE':EICE,'EICE2':EICE2, 'llf_test_model':llf_test_model, 'llf_train':llf_train, 'llf_boot':llf_boot, 'llf_test':llf_test, 'EIC_aic':EIC_aic, 'EIC_bic':EIC_bic,'complexity':len(actual_fit.beta)}
